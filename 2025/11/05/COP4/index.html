设计草稿

总体设计图如下：

erDiagram
    mips {
        input clk
        input reset
    }
    mips ||--o{ pc : clk_rest
    mips ||--o{ grf : clk_rest
    mips ||--o{ mem : clk_rest


    pc {
        input clk
        input reset
        input[31_0] npc
        output[31_0] adr
    }

    pc ||--o{ npc : adr
    pc ||--o{ im : adr
    pc ||--o{ muxGrfWdata : adr
    pc ||--o{ grf : adr
    pc ||--o{ mem : adr


    npc {
        input logicOutput
        input[2_0]  mode
        input[31_0] pc
        input[31_0] immSignExt
        input[25_0] instr_index
        input[31_0] aluOut
        output[31_0] npc
    }

    npc ||--o{ pc : npc
    

    im {
        input[31_0] adr
        output[31_0] instr
    }

    im ||--o{ spilter : instr

    spilter {
        input[31_0] instr
        output[31_26] opcode
        output[25_21] rs
        output[20_16] rt
        output[15_11] rd
        output[10_6] shamt
        output[5_0] func
        output[15_0] imm
        output[25_0] instr_index
    }

    spilter ||--o{ controller : opcode_func
    spilter ||--o{ muxGrfRd1 : rs
    spilter ||--o{ muxGrfRd2 : rt
    spilter ||--o{ muxGrfWt : rd
    spilter ||--o{ muxGrfWt : rt
    spilter ||--o{ ext : imm
    spilter ||--o{ npc : instr_index

    controller {
        input[5_0] opcode
        input[5_0] func 
    }

    controller ||--o{ muxGrfRd1 : rd1Choose
    controller ||--o{ muxGrfRd2 : rd2Choose
    controller ||--o{ muxGrfWt : wtChoose
    controller ||--o{ muxGrfWdata : wdataChoose
    controller ||--o{ muxA : AChoose
    controller ||--o{ muxB : BChoose
    controller ||--o{ muxMemAdr : memAdrChoose
    controller ||--o{ muxMemWdata : memWdataChoose
    controller ||--o{ alu : aluOp
    controller ||--o{ grf : grfWE
    controller ||--o{ mem : memWrite
    controller ||--o{ npc : mode

    muxGrfRd1 {
        input[2_0] rd1Choose
        input[5_0] rs
        output[5_0] rd1
    }

    muxGrfRd1 ||--o{ grf : rd1

    muxGrfRd2 {
        input[2_0] rd2Choose
        input[5_0] rt
        output[5_0] rd2
    }    

    muxGrfRd2 ||--o{ grf : rd2

    muxGrfWt {
        input[2_0] wtChoose
        input[5_0] rd
        input[5_0] rt
        output[5_0] wt
    }    

    muxGrfWt ||--o{ grf : wt

    muxGrfWdata {
        input[2_0] wdataChoose
        input[31_0] aluOut
        input[31_0] memOut
        input[31_0] pc
        output[31_0] wdata
    }

    muxGrfWdata ||--o{ grf : wdata

    grf {
        input clk
        input reset
        input[4_0] rd1
        input[4_0] rd2
        input[4_0] wt
        input WE
        input[31_0] wdata
        input[31_0] wPc
        output[31_0] rdata1
        output[31_0] rdata2
    }

    grf ||--o{ muxA : rdata1
    grf ||--o{ muxB : rdata2
    grf ||--o{ muxMemWdata : rdata2


    alu {
        input[5_0] aluOp
        input[31_0] A
        input[31_0] B
        output[31_0] output
        output logicOutput
    }

    alu ||--o{ muxGrfWdata : aluOut
    alu ||--o{ muxMemAdr : aluOut
    alu ||--o{ npc : aluOut
    alu ||--o{ npc : logicOutput

    muxA {
        input[2_0] Achoose
        input[31_0] GPR[rs]
        output[31_0] A
    }

    muxA ||--o{ alu : A

    muxB {
        input[2_0] Bchoose
        input[31_0] GPR[rt]
        input[31_0] immZeroExt
        input[31_0] immSignExt
        output[31_0] B
    }

    muxB ||--o{ alu : B    

    muxMemAdr {
        input[2_0] memAdrChoose
        input[31_0] aluOut
        output[31_0] memAdr
    }

    muxMemAdr ||--o{ mem : memAdr

    muxMemWdata {
        input[2_0] memWdataChoose
        input[31_0] GPR[rt]
        output[31_0] memWdata
    }

    muxMemWdata ||--o{ mem : memWdata

    mem {
        input clk
        input reset
        input[31_0] memAdr
        input memWrite
        input[31_0] wdata
        input[31_0] wPc
        output[31_0] memOut
    }

    mem ||--o{ muxGrfWdata : memOut

    ext {
        input[15_0] imm
        output[31_0] immSignExt
        output[31_0] immZeroExt
    }

    ext ||--o{ muxB : immZeroExt
    ext ||--o{ muxB : immSignExt 
    ext ||--o{ npc : immSignExt

总的来说，摒弃了P3中权责暧昧的设计，将所有控制信号完全交由controller进行控制，其他模块只管根据选择信号进行相应操作即可

模块设计

mux

为了便于扩展，每个选择器都设置为三位选择信号，由controller统一发出

pc and im

def name
input clk
input reset
input[31:0] npc
output[31:0] pc

im需要注意：

npc

def name
input logicOut
input[2:0] mode
input[31:0] pc
input[31:0] immSignExt
input[25:0] instr_index
input[31:0] aluOut
output[31:0] npc
mode function relevant
000 npc = pc + 4 -
001 logic jump mode beq
010 j instr mode jal
011 jr mode jr

jr、jalr 可以看成是 add 指令的延展，因为指令中的 rt 都是 $0 ,做加法后不影响结果，jalr 还可以直接合并数据通路写入 rd

splitter

按照 R 、 I 、 J 三种方法区分出所有的信号即可

grf

Notice：

不修改 $0

同步复位，上升沿修改，同时输出修改信息

alu

aluOp function
000000 A + B(unsigned)
000001 A - B(unsigned)
000010 A or B
000011 {B[15:0],A[15:0]}
000100 A == B ?

mem

和 grf 几乎一致，只需要注意没有 $0 的限制，输出格式略有不同即可

controller

吐槽：

这才是真正的 core 吧

首先从外部分析，需要对所有的mux发出choose信号，对alu发出op信号，对npc发出mode信号，对 grf 和 mem 发出写入信号（这样分类比较简单，并且完全将解析指令交给 controller 完成）

然后就是根据指令集在数据流向路径上发出对应信号即可，不过要注意对每个部分保留缺省值，如写入信号默认为 0 ，mode 默认为 0（主要是有存储单元的部分），这样可以保证不会产生指令交叉

思考：

到流水线cpu中，存储单元暴增，或许可能会忘记设置缺省值，一定要小心
